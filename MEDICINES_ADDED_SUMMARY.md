### Verification Queries:

```sql
-- Check total medicines
SELECT COUNT(*) FROM medicine WHERE status = 1;
-- Result: 120

-- Check medicines per category
SELECT 
    c.id,
    c.name,
    COUNT(cm.medicine_id) as medicine_count
FROM category c
LEFT JOIN category_medicine cm ON c.id = cm.category_id
GROUP BY c.id, c.name;
-- Result: Each category has 120 medicines

-- Check medicine brands
SELECT * FROM medicine_brands;
-- Result: 6 brands

-- View medicines in a category
SELECT 
    m.id,
    m.name,
    m.strength,
    m.form,
    mb.name as brand_name
FROM medicine m
JOIN category_medicine cm ON m.id = cm.medicine_id
JOIN medicine_brands mb ON m.brand_id = mb.id
WHERE cm.category_id = 4
ORDER BY m.name;
-- Result: 120 medicines for Hairfall category
```
